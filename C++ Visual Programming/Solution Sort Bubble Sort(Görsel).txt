private: System::Void button1_Click(System::Object^  sender, System::EventArgs^  e) {
	int a = System::Convert::ToInt64(textBox1->Text);
	int b[30];
	int temp = a;
	int i = 0;
	int k;
	int gecici;
	int min;
	int index;
	while (temp != 0)
	{
		b[i] = temp % 10;
		temp = temp / 10;
		i++;
	}
	/*for (i = 0; i < 10; i++) (Bubble Sort)
	{
		for (k = 0; k < 10 - 1 - i; k++)
		{
			if(b[k]>b[k+1])
			{
				gecici = b[k];
				b[k] = b[k+1];
				b[k+1] = gecici;
			}
		}
	}*/

	for (i = 0; i < 10 - 1; i++)//Selection Sort
	{
		min = b[10 - 1];
		index = 10 - 1;
		for (k = i; k < 10 - 1; k++)
		{
			if (min > b[k])
			{
				min = b[k];
				index = k;
			}
		}
		b[index] = b[i];
		b[i] = min;
	}




	for (i = 0; i < 10; i++)
	{
		listBox1->Items->Add(System::Convert::ToString(b[i]));
	}
}
